set(prefix "${MODULE}")

add_test(
    NAME ${prefix}_can_Run
    COMMAND ${APPLICATION}
)

    LABELS "${MODULE}")

    NAME ${prefix}_first_test
    COMMAND ${APPLICATION} 1 10
)

    PASS_REGULAR_EXPRESSION "1"
    LABELS "${MODULE}")
